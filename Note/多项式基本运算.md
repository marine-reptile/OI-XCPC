#### 多项式求逆

考虑将问题分治，将当前的多项式的前 $n$ 项和后 $n$ 项分别设为 $f_0$ 和 $f$，

考虑当前有两个多项式，分别是 $f_0+f$ 和 $g_0+g$。

如果我们知道 $f_0*g_0\equiv 1 \pmod {x^{n}}$

那么
$$
\begin{equation}
\begin{split}
(f+f_0)(g+g_0) &\equiv 1 &\pmod {x^{2n}}\\
fg+fg_0+gf_0+f_0g_0 &\equiv 1 &\pmod {x^{2n}}\\
gf_0&\equiv 1-fg_0-f_0g_0&\pmod {x^{2n}}\\
(gf_0)g_0&\equiv g_0(1-fg_0-f_0g_0)&\pmod{x^{2n}}
\end{split}
\end{equation}
$$
因为 $g\equiv 0\pmod{x^{n}}$，
$$
\begin{equation}
\begin{split}
g&\equiv g(f_0g_0)&\pmod{x^{2n}}\\
&\equiv g_0(1-(f+f_0)g_0)&\pmod{x^{2n}}\\
g_0+g&\equiv g_0(2-(f+f_0)g_0)&\pmod{x^{2n}}
\end{split}
\end{equation}
$$

如果 $f*g\equiv 1\pmod {x^{2^k}}(2^k>n)$，
$$
f*g\equiv 1\pmod {x^n}
$$

故只要求出 $\bmod x^{2^k}$ 下的结果即可。



#### 多项式开方

考虑将问题分治 
$$
\begin{equation}
\begin{split}
(f_0)^2&\equiv g&\pmod {x^n}\\
f_0^2-g&\equiv 0&\pmod{x^n}\\
(f_0^2-g)^2&\equiv 0&\pmod{x^{2n}}\\
(f_0^2+g)^2&\equiv 4f_0^2g&\pmod{x^{2n}}\\
(\dfrac{f_0^2+g}{2f_0})^2&\equiv g&\pmod {x^{2n}}
\end{split}
\end{equation}
$$
注意对于常数项而言，需要满足 $g_0$ 存在二次剩余，且需要求出其二次剩余。



#### 多项式带余除法

给定 $f(x),g(x)$，求 $s(x),r(x)$ 使得 $f(x)=g(x)s(x)+r(x)$，$f(x)$ $g(x)$ 的次数分别是 $n-1$ 和 $m-1$

注意性质，$r(x)$ 的次数是 $m-2$，$s(x)$ 的次数是 $n-m$

考虑变换 $f^R(x)=x^{\deg {f}} f(\dfrac{1}{x})$，即反转 $f$ 的系数
$$
\begin{equation}
\begin{split}
f(x)&=g(x)s(x)+r(x)\\
f^R(x)&=(g(x)s(x)+r(x))^R\\
x^{n-1}f(\dfrac{1}{x})&=x^{n-1}g(\dfrac{1}{x})s(\dfrac{1}{x})+x^{n-1}r(\dfrac{1}{x})\\
f^R(x)&=g^R(x)s^R(x)+x^{n-m+1}r^R(x)

\end{split}
\end{equation}
$$
因为 $s^R(x)$ 的次数为 $n-m$，在模 $x^{n-m+1}$ 的意义下求出的 $s^R(x)$ 是符合要求的



#### 多项式牛顿迭代

求 $f(x)$ 使得 $g(f(x))\equiv 0\pmod {x^n}$

考虑如果有 $g(f_0(x))\equiv 0\pmod {x^n}$

在 $f_0(x)$ 处展开 $g(f(x))$
$$
g(f(x))=\sum_{i=0}^\infty \dfrac{g^{(i)}(f_0(x))}{i!}(f(x)-f_0(x))^i
$$
因为 $f(x)-f_0(x)$ 的最低次项是 $x^n$，所以对于 $i\ge 2$，有 $(f(x)-f_0(x))^i\equiv 0\pmod {x^{2n}}$
$$
\begin{equation}
\begin{split}
g(f(x))&\equiv g(f_0(x))+g'(f_0(x))(f(x)-f_0(x))&\pmod {x^{2n}}\\
g(f_0(x))+g'(f_0(x))(f(x)-f_0(x))&\equiv 0&\pmod {x^{2n}}\\
f(x)&\equiv f_0(x)-\dfrac{g(f_0(x))}{g'(f_0(x))}
\end{split}
\end{equation}
$$
牛顿迭代可以解决得到多数多项式基本运算的算法。



#### 多项式 $\ln$

$$
\ln f(x)=\int df(x)=\int \dfrac{f'(x)}{f(x)}dx
$$

注：根据定义，只有 $f_0=1$ 的多项式可以求 $\ln$，且所有 $\ln f(x)$ 的结果常数项为 $0$。



#### 多项式 $\exp$

$$
\begin{equation}
\begin{split}
(\exp f(x))'&=\exp f(x)f'(x)\\
[x^n](\exp f(x))'&=\sum_{i=0}^n [x^i]\exp f(x)[x^{n-i}]f'(x)\\
[x^{n+1}](n+1)\exp f(x)&=\sum [x^i]\exp f(x)[x^{n-i+1}](n-i+1)f(x)
\end{split}
\end{equation}
$$

分治 $fft$ 求解出 $\sum_{i=0}^n n([x^{n}]\exp) f(x)x^n$ 即可，复杂度 $O(n\log^2 n)$



考虑使用牛顿迭代
$$
\begin{equation}
\begin{split}
\ln (\exp h(x))&\equiv h(x)&\pmod {x^n}\\
\ln (f(x))-h(x)&\equiv 0&\pmod {x^n}\\
g(f(x))&\equiv \ln(f(x))-h(x)&\pmod {x^n}\\
g'(f(x))&\equiv \dfrac{1}{f(x)}&\pmod {x^n}\\
f(x)&\equiv f_0(x)-\dfrac{\ln(f_0(x))-h(x)}{\dfrac{1}{f_0(x)}}&\pmod {x^n}\\
&\equiv f_0(x)(1-\ln(f_0(x))+h(x))&\pmod {x^n}
\end{split}
\end{equation}
$$



注：根据定义，只有 $f_0=0$ 的多项式可以求 $\exp$，且所有 $\exp f(x)$ 的结果常数项为 $1$。



#### 多项式快速幂

$$
(f(x))^k=(\exp (\ln f(x)))^k=\exp(k\ln f(x))
$$

注意只有 $f_0=1$ 的多项式可以求 $\ln$，但给定的多项式不一定满足此形式，需要先提出常数项单独处理。对于常数项为 $0$ 的多项式，需要提出 $x^a$ 单独处理直到 $f(x)/x^a$ 的常数项不为 $0$。

另：对多项式 $f(x)$，在 $\mathbb Z/m\mathbb Z$ 上，$(f(x))^k\equiv (f(x))^{m\bmod P} $，但是常数项满足的却是 $(a)^k\equiv (a)^{k\bmod \phi(m)}$。



#### 多项式三角函数

由欧拉公式：
$$
e^{ix}=\cos x+i\sin x
$$
取反 $x$ 得：
$$
e^{-ix}=\cos x-i\sin x
$$
故
$$
\sin x=\dfrac{e^{ix}-e^{-ix}}{2i}\\
\cos x=\dfrac{e^{ix}+e^{-ix}}{2}
$$

$$
\sin f(x)=\dfrac{\exp(if(x))-\exp (-if(x))}{2i}\\
\cos f(x)=\dfrac{\exp(if(x))+\exp (-if(x))}{2}
$$

注意，在剩余类环 $\mathbb Z/m\mathbb Z$ 的意义下，$i$ 的定义为 $m-1$ 在 $\mathbb Z/m\mathbb Z$ 下的二次剩余。

如 $m=998244353$，则 $i=86583718$ 或 $911660535$。



#### 多项式反三角函数

考虑对反三角函数求导，可得
$$
\begin{equation}
\begin{split}
\arcsin x &=\int \dfrac{1}{\sqrt{1-x^2}}\mathrm d x\\
\arccos x &=-\int \dfrac{1}{\sqrt{1-x^2}}\mathrm d x\\
\arctan x &=\int \dfrac{1}{1+x^2}\mathrm d x
\end{split}
\end{equation}
$$
然后代入得到
$$
\begin{equation}
\begin{split}
\arcsin f(x)&=\int\dfrac{1}{\sqrt{1-f^2(x)}}\mathrm d f(x)\\ &=\int\dfrac{f'(x)}{\sqrt{1-f^2(x)}}\mathrm d x\\
\arccos f(x)&=-\int \dfrac{1}{\sqrt{1-f^2(x)}}\mathrm d f(x)\\ &=-\int \dfrac{f'(x)}{\sqrt{1-f^2(x)}}\mathrm d x\\
\arctan f(x) &=\int \dfrac{1}{1+f^2(x)}\mathrm d f(x)\\ &=\int \dfrac{f'(x)}{1+f^2(x)}\mathrm d x
\end{split}
\end{equation}
$$


对照着求就可以了。

namespace polynomial{

    const int N=1048576;
    const int P=998244353;
    namespace num{
        inline int fpow(int a,int p){
            int res=1,cur=a;
            for(int i=0;i<=30;i++){
                if(p>>i&1)res=1ll*res*cur%P;
                cur=1ll*cur*cur%P;
            }return res;
        }
        inline int inv(int a){
            return fpow(a,P-2);
        }
    }using namespace num;

    namespace multiply{
        inline int tobinary(int x){
            return 1<<(__lg(x-1)+1);
        }
        int rvs[N];
        inline void change(int *f,int len){
            for(int i=0;i<len;i++){
                rvs[i]=rvs[i>>1]>>1;
                if(i&1)rvs[i]|=(len>>1);
            }
            for(int i=0;i<len;i++)if(i<rvs[i]){
                swap(f[i],f[rvs[i]]);
            }
        }
        inline void ntt(int *f,int len,int fl){
            change(f,len);
            for(int p=2;p<=len;p<<=1){
                ll syf=fpow(3,(P-1)/p);
                for(int i=0;i<len;i+=p){
                    ll cyr=1;
                    for(int j=i;j<i+p/2;j++){
                        ll g=f[j],t=cyr*f[j+p/2]%P;
                        f[j]=(g+t)%P,f[j+p/2]=(g-t+P)%P;
                        cyr=cyr*syf%P;
                    }
                }
            }if(fl==-1){
                ll iv=inv(len);
                reverse(f+1,f+len);
                for(int i=0;i<len;i++){
                    f[i]=f[i]*iv%P;
                }
            }
        }
        int multiF[N<<1],multiG[N<<1];
        inline void multi(int *F,int n,int *G,int m,int *X,int tom=0){
            int siz=tobinary(n+m-1);
            memset(multiF,0,siz*sizeof(int));
            memset(multiG,0,siz*sizeof(int));
            memcpy(multiF,F,n*sizeof(int));
            memcpy(multiG,G,m*sizeof(int));
            ntt(multiF,siz,1);
            ntt(multiG,siz,1);
            for(int i=0;i<siz;i++){
                multiF[i]=1ll*multiF[i]*multiG[i]%P;
            }
            ntt(multiF,siz,-1);
            memcpy(X,multiF,(tom?tom:siz)*sizeof(int));
        }
    }using namespace multiply;

    namespace inversion{
        int invG[N<<1],invT[N<<1];
        inline void inv(int *F,int n,int *X){
            invG[0]=num::inv(F[0]);
            for(int len=2;(len>>1)<n;len<<=1){
                multi(F,min(n,len),invG,len>>1,invT);
                for(int i=0;i<len;i++){
                    invT[i]=(P-invT[i])%P;
                }
                invT[0]=(invT[0]+2)%P;
                multi(invT,len,invG,len>>1,invG,len);
            }
            memcpy(X,invG,n*sizeof(int));
        }
    }using inversion::inv;
    
    namespace squareroot{
        int sqrG[N<<1],sqrF[N<<1],sqrH[N<<1];
        inline void sqrt(int *F,int n,int *X){
            int tot=tobinary(n);
            memset(sqrG,0,tot*sizeof(int));
            sqrG[0]=1;
            int tm=num::inv(2);
            for(int len=2;(len>>1)<n;len<<=1){
                memcpy(sqrF,F,len*sizeof(int));
                inv(sqrG,len,sqrH);
                ntt(sqrF,len<<1,1);
                ntt(sqrH,len<<1,1);
                for(int i=0;i<(len<<1);i++){
                    sqrH[i]=1ll*sqrF[i]*sqrH[i]%P;
                }
                ntt(sqrH,len<<1,-1);
                for(int i=0;i<len;i++){
                    sqrG[i]=1ll*(sqrG[i]+sqrH[i])%P*tm%P;
                }
            }
            memcpy(X,sqrG,n*sizeof(int));
        }
    }using squareroot::sqrt;

    namespace division{
        int divF[N],outM[N],outR[N];
        inline void divid(int *F,int n,int *G,int m,int *M,int *R){
            reverse(F,F+n);
            reverse(G,G+m);
            inv(G,n-m+1,divF);
            multi(F,n,divF,n-m+1,outM,n-m+1);
            multi(G,m,outM,n-m+1,outR,n);
            for(int i=0;i<m-1;i++){
                outR[i]=(F[i]-outR[i+(n-m+1)]+P)%P;
            }
            reverse(outM,outM+n-m+1);
            reverse(outR,outR+m-1);
            memcpy(M,outM,(n-m+1)*sizeof(int));
        }
    }using division::divid;

    namespace derivation{
        int defF[N<<1];
        inline void def(int *F,int n,int *G){
            for(int i=0;i<n-1;i++){
                defF[i]=1ll*(i+1)*F[i+1]%P;
            }
            defF[n-1]=0;
            memcpy(G,defF,n*sizeof(int));
        }
    }using derivation::def;

    namespace integral{
        int ivt[N<<1];
        inline void inte(int *F,int n,int *G){
            ivt[0]=ivt[1]=1;
            for(int i=2;i<n;i++){
                ivt[i]=1ll*(P-P/i)*ivt[P%i]%P;
            }
            for(int i=1;i<n;i++){
                ivt[i]=1ll*ivt[i]*F[i-1]%P;
            }
            ivt[0]=0;
            memcpy(G,ivt,n*sizeof(int));
        }
    }using integral::inte;

    namespace logarithm{
        int lnF[N<<1],lnG[N<<1];
        inline void ln(int *F,int n,int *G){
            memcpy(lnF,F,n*sizeof(int));
            memcpy(lnG,F,n*sizeof(int));
            inv(lnF,n,lnF);
            def(lnG,n,lnG);
            multi(lnF,n,lnG,n,lnF,n);
            inte(lnF,n,lnF);
            memcpy(G,lnF,n*sizeof(int));
        }
    }using logarithm::ln;

    namespace exponential{
        int expF[N<<1],expG[N<<1];
        inline void exp(int *F,int n,int *G){
            int tot=tobinary(n);
            memset(expF,0,tot*sizeof(int));
            memset(expG,0,tot*sizeof(int));
            expF[0]=1;
            for(int len=2;(len>>1)<n;len<<=1){
                ln(expF,len,expG);
                for(int i=0;i<len;i++){
                    expG[i]=(P+(i==0)-expG[i]+(i<n?F[i]:0))%P;
                }
                multi(expF,len,expG,len,expF,len);
            }
            memcpy(G,expF,n*sizeof(int));
        }
    }using exponential::exp;

    namespace fastpower{
        int pwF[N];
        inline void fpow(int *F,int n,int k,int *G){
            memcpy(pwF,F,n*sizeof(int));
            ln(pwF,n,pwF);
            for(int i=0;i<n;i++){
                pwF[i]=1ll*pwF[i]*k%P;
            }
            exp(pwF,n,pwF);
            memcpy(G,pwF,n*sizeof(int));
        }
    }using fastpower::fpow;

    namespace trigonometric{
        const int I=86583718;
        int triF[N],triG[N];
        inline void sin(int *F,int n,int *G){
            int d=num::inv(2ll*I%P);
            memcpy(triF,F,n*sizeof(int));
            for(int i=0;i<n;i++){
                triF[i]=1ll*triF[i]*I%P;
                triG[i]=(P-triF[i])%P;
            }
            exp(triF,n,triF);
            exp(triG,n,triG);
            for(int i=0;i<n;i++){
                triF[i]=1ll*(triF[i]-triG[i]+P)%P*d%P;
            }
            memcpy(G,triF,n*sizeof(int));
        }
        inline void cos(int *F,int n,int *G){
            int d=num::inv(2);
            memcpy(triF,F,n*sizeof(int));
            for(int i=0;i<n;i++){
                triF[i]=1ll*triF[i]*I%P;
                triG[i]=(P-triF[i])%P;
            }
            exp(triF,n,triF);
            exp(triG,n,triG);
            for(int i=0;i<n;i++){
                triF[i]=1ll*(triF[i]+triG[i])%P*d%P;
            }
            memcpy(G,triF,n*sizeof(int));
        }
    }
    using trigonometric::sin;
    using trigonometric::cos;
    namespace arctrigonometric{
        int arcF[N],arcG[N];
        inline void arcsin(int *F,int n,int *G){
            def(F,n,arcG);
            multi(F,n,F,n,arcF,n);
            for(int i=0;i<n;i++){
                arcF[i]=(P-arcF[i])%P;
            }
            arcF[0]=(arcF[0]+1)%P;
            sqrt(arcF,n,arcF);
            inv(arcF,n,arcF);
            multi(arcF,n,arcG,n,arcF,n);
            inte(arcF,n,G);
        }
        inline void arccos(int *F,int n,int *G){
            def(F,n,arcG);
            multi(F,n,F,n,arcF,n);
            arcF[0]=(arcF[0]-1+P)%P;
            sqrt(arcF,n,arcF);
            inv(arcF,n,arcF);
            multi(arcF,n,arcG,n,arcF,n);
            inte(arcF,n,G);
        }
        inline void arctan(int *F,int n,int *G){
            def(F,n,arcG);
            multi(F,n,F,n,arcF,n);
            arcF[0]=(arcF[0]+1)%P;
            inv(arcF,n,arcF);
            multi(arcF,n,arcG,n,arcF,n);
            inte(arcF,n,G);
        }
    }
    using arctrigonometric::arcsin;
    using arctrigonometric::arccos;
    using arctrigonometric::arctan;
}
